# Ubuntu 24.04 — 本项目依赖（中文说明）

# 更新包索引
sudo apt update

# 1) 编译与构建工具（必须）
sudo apt install -y cmake git

# 2) C++ 库（项目中使用或在 CMake 中查找）
# - nlohmann/json: C++ JSON 库（系统包）
sudo apt install -y nlohmann-json3-dev
# - yaml-cpp: 解析 YAML 配置
sudo apt install -y libyaml-cpp-dev

# 3) 加密/哈希（密码、JWT 等）
sudo apt install -y libargon2-dev libssl-dev

# 4) MySQL 客户端与服务端（开发/本地测试时使用）
sudo apt install -y mysql-server default-libmysqlclient-dev mysql-client

# 5) 单元测试（googletest，编译测试时需要）
sudo apt install -y libgtest-dev

# 6) 常用工具（打包/解压/对比输出）
sudo apt install -y zip unzip diffutils

# 7) 隔离执行（nsjail）
# 项目在判题沙箱中推荐使用 nsjail，但 Ubuntu 24.04 官方仓库通常没有 nsjail 包，
# 建议从源码编译并安装。编译依赖如下：
sudo apt install -y bison flex libprotobuf-dev protobuf-compiler \
	libnl-3-dev libnl-genl-3-dev libnl-route-3-dev libcap-dev libseccomp-dev

# 从源码构建 nsjail（示例）：
rm -rf /tmp/nsjail
git clone --depth 1 https://github.com/google/nsjail.git /tmp/nsjail
make -C /tmp/nsjail -j"$(nproc)"
sudo install -m 0755 /tmp/nsjail/nsjail /usr/local/bin/nsjail

# 说明与注意事项：
# - cpp-httplib 为头文件库（需要 httplib.h 可被编译器找到）；如果未通过系统包提供，
#   可将其作为子模块或把头文件放到项目 include 路径下。
# - 某些系统库（例如 libcpp-httplib）并非标准 apt 包，故未列入自动安装项。
# - 如需在 CI/容器中复现编译环境，可将上述 apt 命令合并到 Dockerfile 或自动化脚本中。
# - 若不需要在本机运行 MySQL 服务，可省略 `mysql-server`，仅安装 `default-libmysqlclient-dev`。
